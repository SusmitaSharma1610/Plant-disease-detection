import matplotlib.pyplot as plt

# Function to preprocess a single image for prediction
def preprocess_image_with_coords(img_path, target_size=(128, 128)):
    img = image.load_img(img_path, target_size=target_size)
    img_array = image.img_to_array(img)
    img_array = np.expand_dims(img_array, axis=0)
    img_array /= 255.0  # Rescale pixel values to [0, 1]
    return img, img_array

# Function to make predictions on a single image and visualize the affected area
def predict_image_with_coords(img_path):
    original_img, preprocessed_img = preprocess_image_with_coords('C:/Users/sushm/OneDrive/Desktop/leaf images/2_5.jpg')
    prediction = model.predict(preprocessed_img)
    class_indices = train_generator.class_indices
    inv_class_indices = {v: k for k, v in class_indices.items()}
    predicted_class = inv_class_indices[np.argmax(prediction)]

    
    width, height = original_img.size
    bbox_coords = (width // 4, height // 4, width // 2, height // 2)

    # Convert PIL Image to NumPy array
    img_with_bbox = np.array(original_img)

    # Draw the rectangle on the original image

    # Display the original image and the image with the bounding box

    plt.title("Original Image")

    plt.subplot(1, 2, 2)

    plt.title("Image with Bounding Box")

    plt.show()

    return predicted_class
